1. Class Hewan (Polimorfisme)

Kode ini mendemonstrasikan polimorfisme dan abstraksi. Kelas abstrak Hewan memiliki metode bersuara() yang belum diimplementasikan. Subkelas seperti Kucing, Anjing, dan Burung kemudian menyediakan implementasi unik untuk bersuara(), menunjukkan perilaku yang berbeda dari metode yang sama.

Kode Dart:


// ================================
// Nama : Yoza Ulpia Rafila
// NIM        : 251410004
// Kelas : SI2A
// ================================

abstract class Hewan {
  String nama = "Hewan";

  void bersuara();
}

class Kucing extends Hewan {
  Kucing() {
    nama = "Kucing";
  }

  @override
  void bersuara() {
    print('$nama: Meong');
  }
}

class Anjing extends Hewan {
  Anjing() {
    nama = "Anjing";
  }

  @override
  void bersuara() {
    print('$nama: Guk guk');
  }
}

class Burung extends Hewan {
  Burung() {
    nama = "Burung";
  }

  @override
  void bersuara() {
    print('$nama: Cuit cuit');
  }
}

void main() {
  print("=== Tugas Polimorfisme &     Abstraksi ===");

  List<Hewan> daftarHewan = [
    Kucing(),
    Anjing(),
    Burung()
  ];

  for (var h in daftarHewan) {
    h.bersuara();
  }
}

2. Class Bentuk (Polimorfisme)

Ini adalah contoh polimorfisme melalui method overriding. Kelas Bentuk memiliki metode gambar() default. Lingkaran dan Persegi menimpa (override) metode ini untuk menampilkan pesan penggambaran yang spesifik untuk bentuk mereka masing-masing.

Kode Dart:


// ================================
// Nama : Yoza Ulpia Rafila
// NIM        : 251410004
// Kelas : SI2A
// ================================

class Bentuk {
  void gambar() {
    print('Menggambar bentuk');
  }
}

class Lingkaran extends Bentuk {
  @override
  void gambar() {
    print('Menggambar lingkaran');
  }
}

class Persegi extends Bentuk {
  @override
  void gambar() {
    print('Menggambar persegi');
  }

  print("=== Polimorfisme Bentuk ===");

  // Menggunakan List (lebih baik &     rapi)
  List<Bentuk> daftarBentuk = [
    Lingkaran(),
    Persegi()
  ];

  for (var bentuk in daftarBentuk) {
    bentuk.gambar();
  }
}

3. Class Kendaraan (Operator `is` & `as`)

Kode ini menunjukkan cara menggunakan operator is untuk memeriksa tipe objek pada saat runtime dan operator as untuk melakukan type casting atau perubahan tipe data. Penggunaan as? untuk safe casting mencegah error jika tipe tidak cocok.

Kode Dart:


/// ================================
/// Nama : Yoza Ulpia Rafila
/// NIM        : 251410004
/// Kelas : SI2A
/// ================================

class Kendaraan {}

class Mobil extends Kendaraan {
  void klakson() {
    print('Tin!');
  }
}

void main() {
  print("=== Operator is dan as ===");
  print("Nama : Yoza Ulpia Rafila");
  print("NIM        : 251410004");
  print("Kelas : SI2A");
  print("--------------------------");

  Kendaraan k = Mobil();

  // Mengecek tipe dengan "is"
  if (k is Mobil) {
    print("Cek tipe: k adalah Mobil");
    k.klakson();
  }

  // Safe casting dengan "as?"
  print("Safe casting:");
  (k as Mobil?)?.klakson();
}

4. Abstract Class Kendaraan

Contoh abstraksi. Kelas Kendaraan dideklarasikan sebagai abstrak dan memiliki metode abstrak start(), yang berarti kelas ini tidak bisa diinstansiasi langsung dan start() harus diimplementasikan oleh kelas turunan (contohnya Mobil). Metode stop() adalah konkret dan memiliki implementasi langsung.

Kode Dart:


/// ================================
/// Nama : Yoza Ulpia Rafila
/// NIM        : 251410004
/// Kelas : SI2A
/// ================================

abstract class Kendaraan {
  String merk;

  // Constructor
  Kendaraan(this.merk);

  // Method abstract
  void start();

  // Method concrete
  void stop() {
    print('Kendaraan berhenti');
  }
}

class Mobil extends Kendaraan {
  // Constructor turunan
  Mobil(String merk) : super(merk);

  @override
  void start() {
    print('Mobil $merk menyala');
  }
}

void main() {
  print("=== Abstract Class Kendaraan ===");
  print("Nama : Yoza Ulpia Rafila");
  print("NIM        : 251410004");
  print("Kelas : SI2A");
  print("--------------------------------");

  Mobil mobil = Mobil("Toyota");

  mobil.start();
  mobil.stop();
}

6. Class Pesawat (Interface)

Menunjukkan konsep interface dalam Dart. Kelas Pesawat mengimplementasikan Kendaraan dan Terbang (yang keduanya adalah kelas abstrak yang bertindak sebagai interface). Ini berarti Pesawat harus menyediakan implementasi untuk semua metode yang didefinisikan dalam kedua "interface" tersebut.

Kode Dart:


/// ================================
/// Nama : Yoza Ulpia Rafila
/// NIM        : 251410004
/// Kelas : SI2A
/// ================================

// Abstract Class
abstract class Kendaraan {
  void jalan();
}

// Interface
abstract class Terbang {
  void terbang();
}

// Class Pesawat
class Pesawat extends Kendaraan implements Terbang {
  @override
  void jalan() {
    print('Pesawat berjalan di landasan');
  }

  @override
  void terbang() {
    print('Pesawat terbang di udara');
  }
}

void main() {
  print("=== OOP Dart ===");
  print("Nama : Yoza Ulpia Rafila");
  print("NIM        : 251410004");
  print("Kelas : SI2A");
  print("--------------------");

  Pesawat pesawat = Pesawat();

  pesawat.jalan();
  pesawat.terbang();
}

9. Class Bentuk Geometri

/// Nama : Yoza Ulpia Rafila

return 2 * 3.14 * radius;

@override
void info() {
print(
'Lingkaran dengan radius $radius memiliki luas ${hitungLuas()} dan keliling ${hitungKeliling()}');
}
}

13. Class Bangun Ruang

Contoh abstraksi dan polimorfisme untuk bangun ruang. Kelas abstrak BangunRuang mendefinisikan metode hitungVolume() dan hitungLuasPermukaan() yang diimplementasikan oleh kelas Kubus dan Balok.

Kode Dart:


/// ================================
/// Nama : Yoza Ulpia Rafila
/// NIM        : 251410004
/// Kelas : SI2A
/// ================================

abstract class BangunRuang {
double hitungVolume();
double hitungLuasPermukaan();
}

class Kubus extends BangunRuang {
double sisi;
Kubus(this.sisi);

@override
double hitungVolume() {
return sisi * sisi * sisi;
}

@override
double hitungLuasPermukaan() {
return 6 * sisi * sisi;
}
}

class Balok extends BangunRuang {
double panjang, lebar, tinggi;
Balok(this.panjang, this.lebar, this.tinggi);

@override
double hitungVolume() {
return panjang * lebar * tinggi;
}

@override
double hitungLuasPermukaan() {
return 2 * ((panjang*lebar) + (panjang*tinggi) + (lebar*tinggi));
}
}

void main() {
print("=== Hitung Bangun Ruang ===");
print("Nama : Yoza Ulpia Rafila");
print("NIM        : 251410004");
print("Kelas : SI2A");
print("--------------------------");

var kubus = Kubus(5);
print('Volume Kubus : ${kubus.hitungVolume()}');
print('Luas Permukaan : ${kubus.hitungLuasPermukaan()}');
}

14. Class Aritmatika

Contoh penggunaan method overloading sederhana dan enkapsulasi. Kelas Aritmatika memiliki metode tambah() yang bisa menerima dua atau tiga parameter, menunjukkan fleksibilitas dalam pemrograman berorientasi objek.

Kode Dart:


/// ================================
/// Nama : Yoza Ulpia Rafila
/// NIM        : 251410004
/// Kelas : SI2A
/// ================================

class Aritmatika {
// Method dengan 2 parameter
int tambah(int a, int b) {
return a + b;
}

// Method dengan 3 parameter (overloading)
int tambahTiga(int a, int b, int c) {
return a + b + c;
}

int kali(int a, int b) {
return a * b;
}
}

void main() {
print("=== Class Aritmatika ===");
print("Nama : Yoza Ulpia Rafila");
print("NIM        : 251410004");
print("Kelas : SI2A");
print("--------------------------");

var hitung = Aritmatika();
print('10 + 5 = ${hitung.tambah(10, 5)}');
print('10 + 5 + 3 = ${hitung.tambahTiga(10, 5, 3)}');
print('10 x 5 = ${hitung.kali(10, 5)}');
}
