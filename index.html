1. Class Hewan (Polimorfisme)

Kode ini mendemonstrasikan polimorfisme dan abstraksi. Kelas abstrak Hewan memiliki metode bersuara() yang belum diimplementasikan. Subkelas seperti Kucing, Anjing, dan Burung kemudian menyediakan implementasi unik untuk bersuara(), menunjukkan perilaku yang berbeda dari metode yang sama.

Kode Dart:

  
// ================================  
// Nama : Yoza Ulpia Rafila  
// NIM        : 251410004  
// Kelas : SI2A  
// ================================  abstract class Hewan {
String nama = "Hewan";

void bersuara();
}

class Kucing extends Hewan {
Kucing() {
nama = "Kucing";
}

@override
void bersuara() {
print('$nama: Meong');
}
}

class Anjing extends Hewan {
Anjing() {
nama = "Anjing";
}

@override
void bersuara() {
print('$nama: Guk guk');
}
}

class Burung extends Hewan {
Burung() {
nama = "Burung";
}

@override
void bersuara() {
print('$nama: Cuit cuit');
}
}

void main() {
print("=== Tugas Polimorfisme &     Abstraksi ===");

List daftarHewan = [
Kucing(),
Anjing(),
Burung()
];

for (var h in daftarHewan) {
h.bersuara();
}
}

2. Class Bentuk (Polimorfisme)

Ini adalah contoh polimorfisme melalui method overriding. Kelas Bentuk memiliki metode gambar() default. Lingkaran dan Persegi menimpa (override) metode ini untuk menampilkan pesan penggambaran yang spesifik untuk bentuk mereka masing-masing.

Kode Dart:

  
// ================================  
// Nama : Yoza Ulpia Rafila  
// NIM        : 251410004  
// Kelas : SI2A  
// ================================  class Bentuk {
void gambar() {
print('Menggambar bentuk');
}
}

class Lingkaran extends Bentuk {
@override
void gambar() {
print('Menggambar lingkaran');
}
}

class Persegi extends Bentuk {
@override
void gambar() {
print('Menggambar persegi');
}
}

void main() {
print("=== Polimorfisme Bentuk ===");

// Menggunakan List (lebih baik &     rapi)
List daftarBentuk = [
Lingkaran(),
Persegi()
];

for (var bentuk in daftarBentuk) {
bentuk.gambar();
}
}

3. Class Kendaraan (Operator `is` & `as`)

Kode ini menunjukkan cara menggunakan operator is untuk memeriksa tipe objek pada saat runtime dan operator as untuk melakukan type casting atau perubahan tipe data. Penggunaan as? untuk safe casting mencegah error jika tipe tidak cocok.

Kode Dart:

  
/// ================================  
/// Nama : Yoza Ulpia Rafila  
/// NIM        : 251410004  
/// Kelas : SI2A  
/// ================================  class Kendaraan {}

class Mobil extends Kendaraan {
void klakson() {
print('Tin!');
}
}

void main() {
print("=== Operator is dan as ===");
print("Nama : Yoza Ulpia Rafila");
print("NIM        : 251410004");
print("Kelas : SI2A");
print("--------------------------");

Kendaraan k = Mobil();

// Mengecek tipe dengan "is"
if (k is Mobil) {
print("Cek tipe: k adalah Mobil");
k.klakson();
}

// Safe casting dengan "as?"
print("Safe casting:");
(k as Mobil?)?.klakson();
}

4. Abstract Class Kendaraan

Contoh abstraksi. Kelas Kendaraan dideklarasikan sebagai abstrak dan memiliki metode abstrak start(), yang berarti kelas ini tidak bisa diinstansiasi langsung dan start() harus diimplementasikan oleh kelas turunan (contohnya Mobil). Metode stop() adalah konkret dan memiliki implementasi langsung.

Kode Dart:

  
/// ================================  
/// Nama : Yoza Ulpia Rafila  
/// NIM        : 251410004  
/// Kelas : SI2A  
/// ================================  abstract class Kendaraan {
String merk;

// Constructor
Kendaraan(this.merk);

// Method abstract
void start();

// Method concrete
void stop() {
print('Kendaraan berhenti');
}
}

class Mobil extends Kendaraan {
// Constructor turunan
Mobil(String merk) : super(merk);

@override
void start() {
print('Mobil $merk menyala');
}
}

void main() {
print("=== Abstract Class Kendaraan ===");
print("Nama : Yoza Ulpia Rafila");
print("NIM        : 251410004");
print("Kelas : SI2A");
print("--------------------------------");

Mobil mobil = Mobil("Toyota");

mobil.start();
mobil.stop();
}

5. Class AlatMusik (Abstraksi)

Kode ini mendemonstrasikan kelas abstrak AlatMusik dengan metode mainkan() yang abstrak (harus diimplementasikan) dan metode info() yang konkret (sudah memiliki implementasi). Kelas Gitar mewarisi dari AlatMusik dan mengimplementasikan mainkan() secara spesifik.

Kode Dart:

  
/// ================================  
/// Nama : Yoza Ulpia Rafila  
/// NIM        : 251410004  
/// Kelas : SI2A  
/// ================================  abstract class AlatMusik {
String nama;

// Constructor
AlatMusik(this.nama);

// Method abstract
void mainkan();

// Method concrete
void info() {
print('Ini adalah alat musik $nama');
}
}

class Gitar extends AlatMusik {
// Constructor turunan
Gitar(String nama) : super(nama);

@override
void mainkan() {
print('Gitar $nama berbunyi: Jreng jreng...');
}
}

void main() {
print("=== Abstract Class Alat Musik ===");
print("Nama : Yoza Ulpia Rafila");
print("NIM        : 251410004");
print("Kelas : SI2A");
print("--------------------------------");

Gitar gitar = Gitar("Akustik");

gitar.info();
gitar.mainkan();
}

6. Class Pesawat (Interface)

Menunjukkan konsep interface dalam Dart. Kelas Pesawat mengimplementasikan Kendaraan dan Terbang (yang keduanya adalah kelas abstrak yang bertindak sebagai interface). Ini berarti Pesawat harus menyediakan implementasi untuk semua metode yang didefinisikan dalam kedua "interface" tersebut.

Kode Dart:

  
/// ================================  
/// Nama : Yoza Ulpia Rafila  
/// NIM        : 251410004  
/// Kelas : SI2A  
/// ================================  // Abstract Class
abstract class Kendaraan {
void jalan();
}

// Interface
abstract class Terbang {
void terbang();
}

// Class Pesawat
class Pesawat extends Kendaraan implements Terbang {
@override
void jalan() {
print('Pesawat berjalan di landasan');
}

@override
void terbang() {
print('Pesawat terbang di udara');
}
}

void main() {
print("=== OOP Dart ===");
print("Nama : Yoza Ulpia Rafila");
print("NIM        : 251410004");
print("Kelas : SI2A");
print("--------------------");

Pesawat pesawat = Pesawat();

pesawat.jalan();
pesawat.terbang();
}

7. Class MathUtils (Static)

Ini adalah contoh penggunaan static member. MathUtils berisi variabel statis (pi) dan metode statis (add). Anggota statis ini dapat diakses langsung melalui nama kelas (MathUtils.pi, MathUtils.add) tanpa perlu membuat objek MathUtils.

Kode Dart:

  
/// ================================  
/// Nama : Yoza Ulpia Rafila  
/// NIM        : 251410004  
/// Kelas : SI2A  
/// ================================  class MathUtils {
// Static variable
static double pi = 3.14;

// Static method
static int add(int a, int b) {
return a + b;
}
}

void main() {
print("=== Static Member ===");
print("Nama : Yoza Ulpia Rafila");
print("NIM        : 251410004");
print("Kelas : SI2A");
print("--------------------------");

// Memanggil langsung lewat nama kelas
print("Nilai PI : ${MathUtils.pi}");
print("Hasil 5 + 3 = ${MathUtils.add(5, 3)}");
}

9. Class Bentuk Geometri

Kode ini menggabungkan abstraksi dan polimorfisme. Kelas abstrak Bentuk mendefinisikan metode untuk hitungLuas() dan hitungKeliling(). Persegi dan Lingkaran adalah kelas konkret yang mewarisi Bentuk dan mengimplementasikan perhitungan sesuai rumus mereka.

Kode Dart:

  
/// ================================  
/// Nama : Yoza Ulpia Rafila  
/// NIM        : 251410004  
/// Kelas : SI2A  
/// ================================  abstract class Bentuk {
double hitungLuas();
double hitungKeliling();
}

class Persegi extends Bentuk {
double sisi;
Persegi(this.sisi);

@override
double hitungLuas() => sisi * sisi;

@override
double hitungKeliling() => 4 * sisi;
}

class Lingkaran extends Bentuk {
double r;
Lingkaran(this.r);

@override
double hitungLuas() => 3.14 * r * r;

@override
double hitungKeliling() => 2 * 3.14 * r;
}

void main() {
print("=== Hitung Geometri ===");
print("Nama : Yoza Ulpia Rafila");
print("NIM        : 251410004");
print("Kelas : SI2A");
print("--------------------------");

var persegi = Persegi(5);
print('Luas Persegi : ${persegi.hitungLuas()}');
print('Keliling Persegi : ${persegi.hitungKeliling()}');
}

12. Polimorfisme List

Ini adalah demonstrasi yang kuat dari polimorfisme di Dart. Objek dari berbagai subkelas (Programmer, Dokter, Guru) dapat disimpan dalam satu list bertipe kelas induk (List<Pekerja>). Saat metode bekerja() dipanggil dalam loop, setiap objek akan memanggil implementasi bekerja() miliknya sendiri.

Kode Dart:

  
/// ================================  
/// Nama : Yoza Ulpia Rafila  
/// NIM        : 251410004  
/// Kelas : SI2A  
/// ================================  abstract class Pekerja {
void bekerja();
}

class Programmer extends Pekerja {
@override void bekerja() { print('Coding...'); }
}

class Dokter extends Pekerja {
@override void bekerja() { print('Mengobati...'); }
}

class Guru extends Pekerja {
@override void bekerja() { print('Mengajar...'); }
}

void main() {
print("=== Polimorfisme List ===");
print("Nama : Yoza Ulpia Rafila");
print("NIM        : 251410004");
print("Kelas : SI2A");
print("--------------------------");

List semuaPekerja = [
Programmer(),
Dokter(),
Guru()
];

for (var p in semuaPekerja) {
p.bekerja();
}
}
